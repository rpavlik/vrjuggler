cmake_minimum_required(VERSION 2.6.2)
project(VRJugglerSuite)

###

set(CPACK_PACKAGE_CONTACT "Ryan Pavlik <rpavlik@iastate.edu>")
set(CPACK_PACKAGE_VERSION_MAJOR "3")
