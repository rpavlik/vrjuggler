	set(TWEEK_INCLUDE_DIRECTORIES ${TWEEK_INCLUDE_DIRECTORIES}

		add_subdirectory(extensions/java)

install(DIRECTORY tweek
	DESTINATION ${TWEEK_INCLUDE_PREFIX}
	COMPONENT tweek-dev
